9pub(crate) fn register_service(
10  connection: &mut PgConnection,
11  service: &Service,
12  corpus: &Corpus,
13  owner: String,
14  description: String,
15) -> Result<(), Error> {
16  use crate::schema::tasks::dsl::*;
17  use crate::schema::{log_errors, log_fatals, log_infos, log_invalids, log_warnings};
18  // The caller resolves the exact corpus (by name) and passes it in — we must NOT re-resolve by
19  // path, because a sandbox shares its parent's path (`find_by_path` would return the parent and
20  // we'd register/check against the wrong corpus). See KNOWN_ISSUES — sandbox activation.
21  let todo_raw = TaskStatus::TODO.raw();
22  let service_id_val = service.id;
23  let corpus_id_val = corpus.id;
24
25  // Idempotent-NEUTRAL guard (defense-in-depth; the HTTP layer also pre-checks and returns 409
26  // without spawning a job). Activation below is destructive — it wipes & re-creates this pair's
27  // tasks *and their `log_*` rows* — so registering a service on a corpus where it is **already
28  // registered** would silently discard completed results. Refuse with no action. A caller that
29  // wants to pick up newly-imported documents uses `extend_service`; one that wants to re-process
30  // uses rerun. (Magic-service note: `import` itself is registered per-corpus and re-importing is a
31  // different path; this guard is for `>2` real services.)
32  let already_registered: i64 = tasks
33    .filter(service_id.eq(service_id_val))
34    .filter(corpus_id.eq(corpus_id_val))
35    .count()
36    .get_result(connection)?;
37  if already_registered > 0 {
38    return Err(Error::QueryBuilderError(
39      format!(
40        "service {:?} is already registered on corpus {:?} ({already_registered} tasks) — \
41         registration is idempotent-neutral; use `extend` to add new documents or `rerun` to \
42         re-process",
43        service.name, corpus.name
44      )
45      .into(),
46    ));
47  }
48
49  // The magic `import` service holds one entry per document; the new service activates over all of
50  // them.
51  let import_service = Service::find_by_name("import", connection)?;
52
53  // (Re)activation atomically clears this <service, corpus> pair's prior tasks **and their `log_*`
54  // rows**, then creates one TODO task per imported entry. The `log_*` tables have no FK to
55  // `tasks` (the only FK is `historical_tasks → tasks ON DELETE CASCADE`), so deleting the tasks
56  // alone would orphan their log rows on every re-activation — the same hazard closed in
57  // `Corpus::destroy`. One transaction so a crash can't leave the service with its tasks deleted
58  // but none re-created.
59  connection.transaction::<(), Error, _>(|t_connection| {
60    let prior_task_ids = || {
61      tasks
62        .filter(service_id.eq(service_id_val))
63        .filter(corpus_id.eq(corpus_id_val))
64        .select(id)
65    };
66    delete(log_infos::table.filter(log_infos::task_id.eq_any(prior_task_ids())))
67      .execute(t_connection)?;
68    delete(log_warnings::table.filter(log_warnings::task_id.eq_any(prior_task_ids())))
69      .execute(t_connection)?;
70    delete(log_errors::table.filter(log_errors::task_id.eq_any(prior_task_ids())))
71      .execute(t_connection)?;
72    delete(log_fatals::table.filter(log_fatals::task_id.eq_any(prior_task_ids())))
73      .execute(t_connection)?;
74    delete(log_invalids::table.filter(log_invalids::task_id.eq_any(prior_task_ids())))
75      .execute(t_connection)?;
76    delete(tasks)
77      .filter(service_id.eq(service_id_val))
78      .filter(corpus_id.eq(corpus_id_val))
79      .execute(t_connection)?;
80    // Create the conversion tasks with ONE server-side `INSERT … SELECT` — no entry list in RAM and
81    // no per-document round-trip (KNOWN_ISSUES I-2). The guard above rejected an already-registered
82    // pair and import entries are unique, so no row conflict is possible here.
83    insert_into(tasks)
84      .values(
85        tasks
86          .filter(service_id.eq(import_service.id))
87          .filter(corpus_id.eq(corpus.id))
88          .select((
89            entry,
90            service_id_val.into_sql::<Integer>(),
91            corpus_id_val.into_sql::<Integer>(),
92            todo_raw.into_sql::<Integer>(),
93          )),
94      )
95      .into_columns((entry, service_id, corpus_id, status))
96      .execute(t_connection)?;
97    // Register the activation run **inside** the same transaction (R-11 sibling), so a crash or a
98    // `mark_new_run` error rolls the freshly-created tasks back too — never
99    // tasks-created-but-no-run, nor a "failed" activate job that in fact created the tasks.
100    // `mark_new_run`'s own transaction nests here as a savepoint.
101    mark::mark_new_run(t_connection, corpus, service, owner, description)
102  })
103}

105pub(crate) fn extend_service(
106  connection: &mut PgConnection,
107  service: &Service,
108  corpus: &Corpus,
109) -> Result<(), Error> {
110  use crate::schema::tasks::dsl::*;
111  // Resolve by the passed corpus, NOT by path: a sandbox shares its parent's path, so
112  // `find_by_path` would return the parent and extend the wrong corpus (same bug as
113  // register_service).
114  let todo_raw = TaskStatus::TODO.raw();
115
116  let import_service = Service::find_by_name("import", connection)?;
117  // Queue a conversion task for every imported document this service doesn't already have one for —
118  // only the genuinely-**new** ones (`ON CONFLICT (entry, service_id, corpus_id) DO NOTHING` is the
119  // bulk twin of the old per-entry `create_if_new`, so re-running `extend` adds the new documents
120  // and leaves existing tasks/results untouched). Done as ONE server-side `INSERT … SELECT`: no
121  // import entry list is materialized in RAM and there is no per-document round-trip, so the work
122  // is bounded regardless of corpus size (KNOWN_ISSUES I-2 — the old path loaded every import
123  // entry into a `Vec` and issued one `create_if_new` per entry, ~1.5M for arxmliv). A single
124  // statement is atomic, so no explicit transaction is needed.
125  insert_into(tasks)
126    .values(
127      tasks
128        .filter(service_id.eq(import_service.id))
129        .filter(corpus_id.eq(corpus.id))
130        .select((
131          entry,
132          service.id.into_sql::<Integer>(),
133          corpus.id.into_sql::<Integer>(),
134          todo_raw.into_sql::<Integer>(),
135        )),
136    )
137    .into_columns((entry, service_id, corpus_id, status))
138    .on_conflict_do_nothing()
139    .execute(connection)?;
140  Ok(())
141}

143/// Permanently destroys a service by name — its definition **plus all tasks + `log_*` rows across
144/// every corpus**, in one transaction (orphan-free + crash-consistent; this *replaces* the latent
145/// `delete_service_by_name`, which deleted only the `services` row and orphaned every dependent
146/// `tasks`/`log_*` row — R-6). Refuses the magic `init` (1) and `import` (2) services, which are
147/// infrastructure (deleting `import` would wipe a corpus's document registry): returns
148/// [`Error::QueryBuilderError`] with a descriptive message rather than touching them. `Err` if the
149/// service is unknown.
150pub(crate) fn destroy_service_by_name(
151  connection: &mut PgConnection,
152  name: &str,
153) -> Result<usize, Error> {
154  let service = Service::find_by_name(name, connection)?;
155  // Defense-in-depth: the frontend already rejects these with 403, but the data layer must never
156  // let a careless caller wipe init/import. (Magic ids: 1=init, 2=import, >2=real services.)
157  if service.id <= 2 {
158    return Err(Error::QueryBuilderError(
159      format!(
160        "refusing to destroy the infrastructure service {:?} (id {})",
161        name, service.id
162      )
163      .into(),
164    ));
165  }
166  service.destroy(connection)
167}
