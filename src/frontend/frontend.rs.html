20// The binary now owns only the static-asset routes; every database-backed route (corpora, reports,
21// runs, services, management, the document-serving + human rerun/save-snapshot paths) lives on the
22// pooled, testable library surface and is mounted by `cortex::frontend::server::mount_api`.
23
24#[get("/favicon.ico")]
25async fn favicon() -> Result<NamedFile, NotFound<String>> {
26  let path = Path::new(&config().assets.public_dir).join("favicon.ico");
27  NamedFile::open(&path)
28    .map_err(|_| NotFound(format!("Bad path: {path:?}")))
29    .await
30}
31
32#[get("/robots.txt")]
33async fn robots() -> Result<NamedFile, NotFound<String>> {
34  let path = Path::new(&config().assets.public_dir).join("robots.txt");
35  NamedFile::open(&path)
36    .map_err(|_| NotFound(format!("Bad path: {path:?}")))
37    .await
38}
39
40#[get("/public/<file..>")]
41async fn files(file: PathBuf) -> Result<NamedFile, NotFound<String>> {
42  let path = Path::new(&config().assets.public_dir).join(file);
43  NamedFile::open(&path)
44    .map_err(|_| NotFound(format!("Bad path: {path:?}")))
45    .await
46}

48#[launch]
49fn rocket() -> _ {
50  // Install our `tracing` subscriber *before* Rocket builds, so application-level events (e.g. the
51  // P-2 slow-report warning in `frontend::concerns`) are actually emitted — Rocket's own subscriber
52  // only surfaces `rocket::*` targets and ignores `RUST_LOG`, so without this the frontend's
53  // app-level `tracing` events go nowhere (which is why legacy code reached for `println!`). Set
54  // first so Rocket finds an existing global subscriber and defers to it (idempotent `try_init`);
55  // `RUST_LOG` now controls frontend verbosity the same way it does the dispatcher.
56  cortex::observability::init_tracing();
57  // Drive the template directory from the runtime configuration rather than a CWD-relative
58  // Rocket.toml, so the binary is not bound to its working directory.
59  let figment =
60    rocket::Config::figment().merge(("template_dir", config().assets.template_dir.as_str()));
61  let rocket = rocket::custom(figment)
62    .mount("/", routes![favicon, robots, files])
63    .attach(CORS());
64  // Mount the library API/UI surface (management/health/settings, corpora, reports, the document +
65  // rerun routes, …); the builder owns the connection pool and the Template fairing.
66  cortex::frontend::server::mount_api(rocket)
67}
